****** utilities
#+begin_src sqlite
CREATE TABLE utility(
comp    TEXT NOT NULL REFERENCES company(id),
short   TEXT,
country NOT NULL REFERENCES country(id),
PRIMARY KEY(comp,country));
#+end_src
